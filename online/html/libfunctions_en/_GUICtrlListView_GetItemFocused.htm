_GUICtrlListView_GetItemFocused

Show description in

Determines whether the item is highlighted as a drag-and-drop target

#include <GuiListView.au3>
_GUICtrlListView_GetItemFocused ( $hWnd, $iIndex )

Parameters

$hWnd Control ID/Handle to the control
$iIndex 0-based index of the item

Return Value

Success: True.
Failure: False.

Related

_GUICtrlListView_SetItemFocused

Example

#include <GUIConstantsEx.au3>
#include <GuiListView.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
        GUICreate("ListView Get/Set Item Focused (v" & @AutoItVersion & ")", 400, 300)
        Local $idListview = GUICtrlCreateListView("", 2, 2, 394, 268)
        GUISetState(@SW_SHOW)

        ; Add columns
        _GUICtrlListView_AddColumn($idListview, "Column 0", 100)
        _GUICtrlListView_AddColumn($idListview, "Column 1", 100)
        _GUICtrlListView_AddColumn($idListview, "Column 2", 100)

        ; Add items
        _GUICtrlListView_AddItem($idListview, "Row 0: Col 0", 0)
        _GUICtrlListView_AddSubItem($idListview, 0, "Row 0: Col 1", 1)
        _GUICtrlListView_AddSubItem($idListview, 0, "Row 0: Col 1", 2)
        _GUICtrlListView_AddItem($idListview, "Row 1: Col 0", 1)
        _GUICtrlListView_AddSubItem($idListview, 1, "Row 1: Col 1", 1)
        _GUICtrlListView_AddItem($idListview, "Row 2: Col 0", 2)

        ; Focus item 1
        _GUICtrlListView_SetItemFocused($idListview, 1)
        MsgBox($MB_SYSTEMMODAL, "Information", "Item 1 Focused: " & _GUICtrlListView_GetItemFocused($idListview, 1))

        ; Loop until the user exits.
        Do
        Until GUIGetMsg() = $GUI_EVENT_CLOSE
        GUIDelete()
EndFunc   ;==>Example
