_GUICtrlListView_CopyItems



Beschreibung anzeigen in

Kopiert Items zwischen 2 ListView-Controls

#include <GuiListView.au3>
_GUICtrlListView_CopyItems ( $hWnd_Source, $hWnd_Destination [, $bDelFlag = False] )

Parameter

$hWnd_Source Control-ID / Handle des Controls der Quelle
$hWnd_Destination Control-ID / Handle des Controls des Ziels
$bDelFlag [optional] nach dem Kopieren löschen (entspricht verschieben)

Rückgabewert

Keine.

Bemerkungen

- - - - - - - - Erklärung der Controls - - - - - - - -

Func _KeepWindowsDocked()
    Local $p_win1 = WinGetPos($g_hGUI1)
    Local $p_win2 = WinGetPos($g_hGUI2)
    If $g_iDock_Location == 1 Then
        If (($p_win1[0] <> $g_iX1 Or $p_win1[1] <> $g_iY1) And BitAND(WinGetState($g_hGUI1), 8) Or $g_iDock = 2) Then
            $g_iX1 = $p_win1[0]
            $g_iY1 = $p_win1[1]
            $g_iX2 = $p_win1[2] + $g_iX1
            $g_iY2 = $g_iY1
            WinMove($g_hGUI2, "", $g_iX2, $g_iY2)
            $g_iDock = 1
        ElseIf (($p_win2[0] <> $g_iX2 Or $p_win2[1] <> $g_iY2) And BitAND(WinGetState($g_hGUI2), 8)) Then
            $g_iX2 = $p_win2[0]
            $g_iY2 = $p_win2[1]
            $g_iX1 = $p_win2[0] - $p_win1[2]
            $g_iY1 = $g_iY2
            WinMove($g_hGUI1, "", $g_iX1, $g_iY1)
        EndIf
    Else
        If (($p_win1[0] <> $g_iX1 Or $p_win1[1] <> $g_iY1) And BitAND(WinGetState($g_hGUI1), 8) Or $g_iDock = 2) Then
            $g_iX1 = $p_win1[0]
            $g_iY1 = $p_win1[1]
            $g_iX2 = $g_iX1
            $g_iY2 = $p_win1[3] + $g_iY1
            WinMove($g_hGUI2, "", $g_iX2, $g_iY2)
            $g_iDock = 1
        ElseIf (($p_win2[0] <> $g_iX2 Or $p_win2[1] <> $g_iY2) And BitAND(WinGetState($g_hGUI2), 8)) Then
            $g_iX2 = $p_win2[0]
            $g_iY2 = $p_win2[1]
            $g_iX1 = $g_iX2
            $g_iY1 = $p_win2[1] - $p_win1[3]
            WinMove($g_hGUI1, "", $g_iX1, $g_iY1)
        EndIf
    EndIf
EndFunc   ;==>_KeepWindowsDocked
